## Install

```bash
npm install --save pokebutton
```

## Usage

```jsx
import React, { Component } from 'react'

import { Button } from 'pokebutton'
import 'pokebutton/dist/index.css'

class Example extends Component {
  render() {
    return <Button />
  }
}
```
